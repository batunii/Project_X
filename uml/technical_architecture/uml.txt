package "I. Client Layer" {
    cloud "VR Headset\n(Meta Horizon OS)" as VR_Headset
    cloud "PC\n(Windows)" as PC

    rectangle "CLIENT-SIDE APPLICATION" {
        component "HUD" as HUD
        component "Gallery Viewer" as GalleryViewer
        component "Art Interaction" as ArtInteraction
        component "User Interaction" as UserInteraction
        component "User Input Manager" as UserInputManager
    }

    note bottom of "CLIENT-SIDE APPLICATION"
      VR ENGINE: Unity
      DEVELOPMENT LANGUAGE: C#
      VR SDKs: Meta XR
    end note

    VR_Headset --> HUD
    PC --> HUD
    HUD -- GalleryViewer
    GalleryViewer -- ArtInteraction
    ArtInteraction -- UserInteraction
    UserInteraction -- UserInputManager
}

    component "DATA TRANSFER: Unity Chat Service(REST APIs)\n" as Network
}

package "III. Backend/Server Layer (Cloud)" {
    cloud "CLOUD PLATFORM:\nUnity Cloud Platform" as CloudPlatform

 
        component "Art Submission Service" as ArtSubmission
        component "Authentication:\n(Supoerbase)" as Auth
        component "Cloud Storage:\n(Unity CDN)" as CloudStorage
        component "Database:\nPostgreSQL" as Database
        component "Logging" as Logging
    

    component "Unity Logging" as Monitoring
    component "CI/CD:\nGitHub Actions / GitLab CI" as CI_CD

  

    CloudPlatform --> ArtSubmission
    CloudPlatform --> Auth
    CloudPlatform --> Logging
    Logging-->Monitoring



    ArtSubmission --> CloudStorage
    Auth --> Database
    CloudPlatform --> CI_CD
}

UserInputManager --> Network
