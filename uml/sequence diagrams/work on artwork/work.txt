@startuml
actor User
participant "Controller/Input Manager" as Input
participant "HUD" as HUD
participant "Canvas" as Canvas
participant "Art Tools" as Tools
participant "Security Manager" as Auth
participant "Event Logger" as Logger
User -> Input :  selects canvas
Input -> Canvas : zooms in
Logger -> Logger: Log: Canvas selected
User -> Input : picks up tool
Input -> Tools : applies tool
Tools -> Canvas : updates canvas
Logger -> Logger: Log: Canvas updated
alt User clicks undo
User -> Input: interacts with undo tool
Input -> Tools: registers interaction
Tools -> Canvas: undoes last stored change
Logger -> Logger: Log: Change undone
end
Canvas -> Auth : requests authentication
Auth -> Canvas: authenticates change
User -> Input : clicks exit
Logger -> Logger: Log: Canvas exited
Input -> HUD : exits canvas
Input -> Canvas : unzooms
@enduml
