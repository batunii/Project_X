@startuml
actor User

participant "HUD" as HUD
participant "Controller/Input Manager" as Input
participant "Auth Manager" as Auth
participant "Profile Manager" as Profile
participant "Room Manager" as Room
participant "Network Manager" as Network
participant "Event Logger" as Event

User -> HUD : Select "Delete Account"
HUD -> Input : Button "Delete Account" clicked
Input -> Auth : Check permissions
Auth -> Input : Reply with permissions

alt Permission granted
Input -> Profile : Request deletion of the account
Profile -> Network : Delete information

alt Account deleted
Network -> Event : Log deletion action
Network -> Room : Load default gallery
Room -> HUD : Change room view
HUD -> User: Display new room
else Account not deleted
Network -> Event : Log deletion attempt
Network -> Input : Unsuccessful attempt
Input -> HUD : Forward error
HUD -> User : Display "Account was not deleted"
end

else Permission denied
Input -> HUD : Deny request
HUD -> User : Display "Deletion not allowed"
end

@enduml
