@startuml
actor "Signed-in User" as User
participant HUD
participant "Input Manager" as Controller
participant "Room Manager" as RoomMgr
participant "Art Tools" as ArtTools

User -> HUD: Selects 'Customize Workspace' option
activate HUD
HUD -> Controller: Opens workspace customization menu
deactivate HUD
activate Controller
Controller -> User: Shows customization options
deactivate Controller

alt Change placement of art pieces
    
    User -> Controller: Selects "Change placement of art pieces"
    activate Controller
    Controller -> RoomMgr: Enter placement edit mode
    deactivate Controller
    activate RoomMgr
    RoomMgr -> User: Enter "Gallery Edit Mode" 
    RoomMgr -> User: Can now move art pieces around
    deactivate RoomMgr
    
else Change Art Supplies
    
    User -> Controller: Selects "Change Art Supplies"
    activate Controller
    Controller -> ArtTools: Enter art supplies edit mode
    deactivate Controller
    activate ArtTools
    ArtTools -> User: Enter "Gallery Edit Mode" 
    ArtTools -> User: Can now modify color palette
    deactivate ArtTools

else Change wall color

    User -> Controller: Selects "Change wall color" 
    activate Controller
    Controller -> RoomMgr: Enter wall color edit mode
    deactivate Controller
    activate RoomMgr
    RoomMgr -> User: Enter "Gallery Edit Mode" 
    RoomMgr -> User: Can now choose wall colors
    deactivate RoomMgr
    
end
@enduml
