@startuml

actor User
participant "Controller/Input Manager" as Controller
participant "Tutorial Manager" as TutorialMgr
participant "Room Manager" as RoomMgr
participant "Event Logger" as Logger

User -> Controller: Clicks 'Tutorial' option (HUD)
Logger -> Logger: Log: Tutorial clicked
Controller -> TutorialMgr: Start tutorial

TutorialMgr -> User: Guides: "How to walk"
User -> Controller: Performs walking action
Controller -> TutorialMgr: Movement completed

TutorialMgr -> User: Guides: "How to zoom in/out"
User -> Controller: Performs zoom action
Controller -> TutorialMgr: Zoom completed

TutorialMgr -> User: Guides: "Painting features (color, save)"
User -> Controller: Performs painting action
Controller -> TutorialMgr: Painting completed

TutorialMgr -> User: Guides: "Access and use the HUD"
User -> Controller: Opens/Explores HUD
Logger -> Logger: Log: HUD Opened
Controller -> TutorialMgr: HUD explored

TutorialMgr -> RoomMgr: Tutorial finished
Logger -> Logger: Log: Tutorial finished
alt Signed-in User
    RoomMgr -> RoomMgr: Load workspace
    RoomMgr -> User: Places user in Workspace (Postcondition 1)
else Guest User
    RoomMgr -> RoomMgr: Load default gallery
    RoomMgr -> User: Places user in Default Gallery (Postcondition 2)
end


@enduml
